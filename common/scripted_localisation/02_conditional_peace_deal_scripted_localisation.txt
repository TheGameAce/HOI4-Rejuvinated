# War Reparations Live Preview — resolves to the projected weekly transfer
# amount. Uses payer's GDP (target if we demand, self if we concede), the
# 0.0005 multiplier, and the £15B/week clamp. Matches CPD_sign_peace_deal
# exactly so the preview never disagrees with the actual transfer.
defined_text = {
	name = CPD_reparations_preview

	text = {
		trigger = {
			if = {
				limit = { ROOT = { has_country_flag = CPD_giving_concessions } }
				set_temp_variable = { CPD_rep_preview_temp = ROOT.gdp_total }
			}
			else = {
				set_temp_variable = { CPD_rep_preview_temp = THIS.gdp_total }
			}
			multiply_temp_variable = { CPD_rep_preview_temp = 0.0005 }
			clamp_temp_variable = { var = CPD_rep_preview_temp max = 15 }
			set_temp_variable = { CPD_rep_preview_total_temp = CPD_rep_preview_temp }
			multiply_temp_variable = { CPD_rep_preview_total_temp = 52 }
		}
		localization_key = "CPD_reparations_preview_TT"
	}
}
