add_namespace = JUNO_brazil_events

country_event = { #Org member is under attack, should we help?
	id = JUNO_brazil_events.130
	title = JUNO_brazil_events.130.t
	desc = JUNO_brazil_events.130.desc
	picture = GFX_report_event_generic_read_write
	is_triggered_only = yes



	option = { # Of course!
		name = JUNO_brazil_events.130.a 
		ai_chance = {
			base = 10
			modifier = {
				factor = 0
				OR = {
					any_other_country = {
						has_war_with = FROM
						is_in_faction_with = ROOT
					}
				}
			}
			modifier = {
				factor = 10
				has_opinion = {
					target = FROM
					value > 40
				}
			}
			modifier = {
				factor = 10
				has_government = FROM
			}
		}
		# FROM = attacker
		# global.org_africa_attacked_country is defender
		# ROOT = called country (i.e. the one getting the event)
		var:global.org_american_unity_attacked_country = {
			ROOT = {
				add_to_war = {
					targeted_alliance = PREV
					enemy = FROM 
					hostility_reason = ally
				}
			}
			country_event = {
				id = JUNO_brazil_events.131
				hours = 1
			}
		}
	}
	option = { # We can not
		name = JUNO_brazil_events.130.b
		ai_chance = {
			base = 3
			modifier = {
				factor = 10
				has_opinion = {
					target = FROM
					value < 0
				}
			}
		}
		var:global.org_american_unity_attacked_country = {
			country_event = {
				id = JUNO_brazil_events.132
				hours = 1
			}
		}
		remove_ideas = SMB_organization_of_american_states_member
	}
}
