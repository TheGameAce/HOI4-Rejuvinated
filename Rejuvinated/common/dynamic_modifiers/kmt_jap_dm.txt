JAP_more_troops = {

	enable = { 
		original_tag = JAP
		is_ai = yes
		has_government = fascism
		any_country = {
			has_war_with = ROOT
			OR = {
				is_major = yes
				num_divisions > 100
			}
		}
		num_divisions < 200
		is_subject = no
		NOT = { has_dynamic_modifier = { modifier = JAP_ai_small_army }}
		NOT = { 
			has_equipment = { infantry_equipment < 7000 }
		}
	}
	
	remove_trigger = { 
		always = no
	}
	
	ai_desired_divisions_factor = 1000
}
