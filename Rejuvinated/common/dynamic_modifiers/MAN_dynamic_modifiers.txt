MAN_local_bandits = {
	icon = GFX_local_bandits
	enable = { 
		CONTROLLER = {
			OR = {
				tag = MAN
				is_in_faction_with = MAN
				is_subject_of = MAN
				MAN = { is_subject_of = PREV }
				tag = JAP
				is_in_faction_with = JAP
				is_subject_of = JAP
				JAP = { is_subject_of = PREV }
			}
		}
	}
	
	remove_trigger = { 
		OR = {
			OWNER = {
				OR = { 
					tag = event_target:KMT_central_gov_tag
					tag = PRC
				}
				is_subject = no
			}
			has_resistance = no
		}
	}
	
	#compliance_gain = MAN_local_bandits_compliance_gain				#-0.01
	resistance_target = MAN_local_bandits_resistance_target			#0.15
	
	compliance_gain = -100
}
