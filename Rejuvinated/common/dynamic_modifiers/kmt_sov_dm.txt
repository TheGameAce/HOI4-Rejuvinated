KMTGER_no_spam = {
	enable = { 
		NOT = { 
			any_country = { 
				has_war_with = ROOT 
				is_ai = no 
			}
			difficulty > 2
		}
		NOT = {
			any_country = {
				has_war_with = ROOT
				check_variable = { THIS.num_divisions > ROOT.num_divisions }
			}
		}
		num_divisions > 250
	}
	
	remove_trigger = { 
		always = no
	}
	
	ai_desired_divisions_factor = -200
}
