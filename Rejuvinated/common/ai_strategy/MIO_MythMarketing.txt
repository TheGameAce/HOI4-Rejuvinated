MythMarketing_ask_ai_buy_equipment_45 = {
    allowed = {
        has_dlc = "Arms Against Tyranny"
    }
    enable = {
        num_of_civilian_factories > 45
		any_country = {
            is_ai = no                            # 确保是玩家
			has_doctrine = Mio_War_Profiteer # 检测玩家是否点了卖装备学说
        }
    }
	abort = {
		OR = {
			num_of_civilian_factories < 45
		}
	}
    abort_when_not_enabled = yes
	# 增加从玩家处购买装备的倾向
    ai_strategy = {
        type = equipment_market_trade_desire                                 
		target_country = PREV  
        value = 9999                               # 高倾向值
    }
    ai_strategy = {
		type = equipment_market_spend_factories
		value = 90
	}
}
